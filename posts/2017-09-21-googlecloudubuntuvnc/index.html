VNC Server on Google Cloud Ubuntu 16.04;

# Updates:
sudo apt-get update
sudo apt-get upgrade

# Setup:
sudo apt install xfce4 xfce4-goodies tightvncserver

vncserver

vncserver -kill :1

mv ~/.vnc/xstartup ~/.vnc/xstartup.bak

nano ~/.vnc/xstartup
xstartup content:
#!/bin/bash
xrdb $HOME/.Xresources
startxfce4 &

sudo chmod +x ~/.vnc/xstartup

vncserver
